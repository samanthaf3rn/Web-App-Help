Basic Structure

  <head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Page Title</title>
  </head>
  <body>
    <h1>Heading</h1>
    <p>Paragraph text here.</p>
  </body>
</html>

Fibonacci

 fibonacci series
function fibb() {
  var n = parseInt(document.form1.n.value);
  var f1 = 0, f2 = 1, f3, i = 0;
  el.innerHTML = f1 + ' ' + f2 + ' ';
  while (i < n - 2) {
    f3 = f1 + f2;
    el.innerHTML += f3 + ' ';
    f1 = f2; f2 = f3; i++;
  }
}

Storage & AJAX

localStorage, sessionStorage, Cookies, and XMLHttpRequest.

 localStorage persistent
// Save
localStorage.setItem(key, message);

// Read
var val = localStorage.getItem(key);

// Remove one key
localStorage.removeItem(key);

// Clear everything
localStorage.clear();

// Note: survives browser/tab close
 sessionStorage tab only
// Same API as localStorage
sessionStorage.setItem(key, message);
var val = sessionStorage.getItem(key);

// Cleared when tab is closed
// Use for temp session data (like cart)
 Cookies (JS)
// Set cookie — 3 day expiry
var date = new Date();
date.setTime(date.getTime() + (3*24*60*60*1000));
document.cookie = "username=" + user
  + "; expires=" + date.toUTCString()
  + "; path=/";

// Read all cookies
var all = document.cookie.split(';');
for (var i = 0; i < all.length; i++) {
  display += all[i].trim() + '<br>';
}
 AJAX — XMLHttpRequest
function loadData() {
  var xhr = new XMLHttpRequest();
  xhr.open('GET', 'data.txt', true);
  xhr.send();

  xhr.onload = function() {
    if (xhr.status === 200) {
      document.getElementById('data')
        .innerHTML = xhr.responseText;
    } else {
      document.getElementById('data')
        .innerHTML = 'Failed to load';
    }
  };
  xhr.onerror = function() {
    document.getElementById('data')
      .innerHTML = 'Error in request';
  };
}
<!-- trigger on keypress -->
<input onkeypress="loadData()">
ℹ Comparison: localStorage = permanent; sessionStorage = tab-scoped; Cookies = can be sent to server with every request (PHP reads via $_COOKIE).

JSON — Parse & Stringify

 JSON.parse — text to object
const txt = '{"name":"samantha","age":20,"city":"dubai"}';
const obj = JSON.parse(txt);
// obj.name → "samantha"
// obj.age  → 20
document.getElementById('demo').innerHTML
  = obj.name + ', ' + obj.age;
 JSON.stringify — object to text
const obj = {
  name: 'samantha',
  age:  20,
  city: 'dubai'
};
const myJSON = JSON.stringify(obj);
// '{"name":"samantha","age":20,"city":"dubai"}'
document.getElementById('demo').innerHTML = myJSON;

Controller — Full CRUD

 StudentController.php
class StudentController extends Controller {

  // ADD
  function add(Request $request) {
    $s = new Student();
    $s->name  = $request->name;
    $s->email = $request->email;
    $s->phone = $request->phone;
    $s->save();
  }

  // LIST
  function list() {
    $data = Student::all();
    return view('list-student', ['student123' => $data]);
  }

  // DELETE
  function deleteStudent($id) {
    Student::where('id', $id)->delete();
    return redirect('listStudent');
  }

  // UPDATE — load form
  function updateStudent($id) {
    $s = Student::find($id);
    return view('updateform', ['studentdata' => $s]);
  }

  // UPDATE — save
  function updateform($id, Request $req) {
    $s = Student::find($id);
    $s->name = $req->name;
    $s->save();
    return redirect('listStudent');
  }
}
